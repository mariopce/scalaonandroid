while

  def main(args: Array[String]) {
    var i = 0

      println("i " + i);
    }

}

Do while
[source]
object ScalaTutorial {
  def main(args: Array[String]) {
    var i = 0
    do {
      println("i " + i);
      i+=1
    } while  (i<10)
  }
}
